‚è∞ Python Alarm Clock

This Python-based alarm clock allows users to set a specific time and plays a sound when the time is reached. It‚Äôs built with the datetime and playsound libraries.

import datetime
import time
import playsound

alarm_time = "07:30"
while True:
    current_time = datetime.datetime.now().strftime("%H:%M")
    if current_time == alarm_time:
        playsound.playsound('alarm.mp3')
        break
    time.sleep(30)

Turn Based battle

a Python turn-based battle game between a player and a computer-controlled monster, with special rules: Player & monster: 100 HP each initially. Choices: Regular attack (available every turn) ‚Äî moderate damage. Strong attack ‚Äî available every 3 turns only ‚Äî high damage. Heal ‚Äî available every 5 turns only ‚Äî restores HP. Damage/heal values random within ranges. Monster attacks after player each turn. If one reaches zero, declare a winner and ask to restart. Here‚Äôs a clean Python implementation for this scenario: 

import random

def battle_game():
    print("= WELCOME TO PLAYER vs MONSTER GAME =")
    while True:
        player_hp = 100
        monster_hp = 100
        turn_counter = 0
        while player_hp > 0 and monster_hp > 0:
            turn_counter += 1
            print(f"\n--- Turn {turn_counter} ---")
            print(f"Player HP: {player_hp} | Monster HP: {monster_hp}")
            # Determine allowed actions
            actions = ["1. Regular Attack"]
            if turn_counter % 3 == 0:
                actions.append("2. Strong Attack")
            if turn_counter % 5 == 0:
                actions.append("3. Heal")
            # Display options
            print("\nChoose your action:")
            for option in actions:
                print(option)
            # Input validation
            choice = input("Enter choice number: ").strip()
            while choice not in [str(i+1) for i in range(len(actions))]:
                choice = input("Invalid choice. Try again: ").strip()
            choice = int(choice)
            # Player action
            if "Regular" in actions[choice-1]:
                dmg = random.randint(10, 20)
                monster_hp -= dmg
                print(f"You hit the monster for {dmg} damage.")
            elif "Strong" in actions[choice-1]:
                dmg = random.randint(25, 40)
                monster_hp -= dmg
                print(f"üí• STRONG ATTACK! You hit the monster for {dmg} damage.")
            elif "Heal" in actions[choice-1]:
                heal_amt = random.randint(15, 30)
                player_hp = min(100, player_hp + heal_amt)
                print(f"‚ú® You healed yourself for {heal_amt} HP.")
            # Check if monster is defeated before it attacks
            if monster_hp <= 0:
                print("\nüéâ You defeated the monster!")
                break
            # Monster's turn
            monster_dmg = random.randint(8, 18)
            player_hp -= monster_dmg
            print(f"The monster hits you for {monster_dmg} damage.")
            # Check if player is defeated
            if player_hp <= 0:
                print("\nüíÄ You were defeated by the monster!")
                break
        # Game over - ask for replay
        again = input("\nDo you want to play again? (y/n): ").lower()
        if again != 'y':
            print("Thanks for playing! Goodbye!")
            break

if __name__ == "__main__":
    battle_game()
