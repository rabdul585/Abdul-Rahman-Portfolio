while True:
    current_time = datetime.datetime.now().strftime("%H:%M")

Turn Based Battle Game

import random

def battle_game():
    print("= WELCOME TO PLAYER vs MONSTER GAME =")
    while True:
        player_hp = 100
        monster_hp = 100
        turn_counter = 0
        while player_hp > 0 and monster_hp > 0:
            turn_counter += 1
            print(f"\\n--- Turn {turn_counter} ---")
            print(f"Player HP: {player_hp} | Monster HP: {monster_hp}")
            actions = ["1. Regular Attack"]
            if turn_counter % 3 == 0:
                actions.append("2. Strong Attack")
            if turn_counter % 5 == 0:
                actions.append("3. Heal")
            print("\\nChoose your action:")
            for option in actions:
                print(option)
            choice = input("Enter choice number: ").strip()
            while choice not in [str(i+1) for i in range(len(actions))]:
                choice = input("Invalid choice. Try again: ").strip()
            choice = int(choice)
            if "Regular" in actions[choice-1]:
                dmg = random.randint(10, 20)
                monster_hp -= dmg
                print(f"You hit the monster for {dmg} damage.")
            elif "Strong" in actions[choice-1]:
                dmg = random.randint(25, 40)
                monster_hp -= dmg
                print(f"üí• STRONG ATTACK! You hit the monster for {dmg} damage.")
            elif "Heal" in actions[choice-1]:
                heal_amt = random.randint(15, 30)
                player_hp = min(100, player_hp + heal_amt)
                print(f"‚ú® You healed yourself for {heal_amt} HP.")
            if monster_hp <= 0:
                print("\\nüéâ You defeated the monster!")
                break
            monster_dmg = random.randint(8, 18)
            player_hp -= monster_dmg
            print(f"The monster hits you for {monster_dmg} damage.")
            if player_hp <= 0:
                print("\\nüíÄ You were defeated by the monster!")
                break
        again = input("\\nDo you want to play again? (y/n): ").lower()
        if again != 'y':
            print("Thanks for playing! Goodbye!")
            break

if __name__ == "__main__":
    battle_game()
    
// Java equivalent example 
import java.util.Scanner;
import java.util.Random;

public class BattleGame {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        Random random = new Random();
        System.out.println("= WELCOME TO PLAYER vs MONSTER GAME =");
        while (true) {
            int playerHp = 100;
            int monsterHp = 100;
            int turnCounter = 0;

            while (playerHp > 0 && monsterHp > 0) {
                turnCounter++;
                System.out.printf("\\n--- Turn %d ---\\n", turnCounter);
                System.out.printf("Player HP: %d | Monster HP: %d\\n", playerHp, monsterHp);
                System.out.println("Choose your action:");
                System.out.println("1. Regular Attack");
                if (turnCounter % 3 == 0) System.out.println("2. Strong Attack");
                if (turnCounter % 5 == 0) System.out.println("3. Heal");
                int choice = scanner.nextInt();

                int dmg, healAmt;
                switch (choice) {
                    case 1:
                        dmg = random.nextInt(11) + 10;
                        monsterHp -= dmg;
                        System.out.printf("You hit the monster for %d damage.\\n", dmg);
                        break;
                    case 2:
                        dmg = random.nextInt(16) + 25;
                        monsterHp -= dmg;
                        System.out.printf("üí• STRONG ATTACK! You hit the monster for %d damage.\\n", dmg);
                        break;
                    case 3:
                        healAmt = random.nextInt(16) + 15;
                        playerHp = Math.min(100, playerHp + healAmt);
                        System.out.printf("‚ú® You healed yourself for %d HP.\\n", healAmt);
                        break;
                    default:
                        System.out.println("Invalid choice.");
                }

                if (monsterHp <= 0) {
                    System.out.println("\\nüéâ You defeated the monster!");
                    break;
                }
                int monsterDmg = random.nextInt(11) + 8;
                playerHp -= monsterDmg;
                System.out.printf("The monster hits you for %d damage.\\n", monsterDmg);

                if (playerHp <= 0) {
                    System.out.println("\\nüíÄ You were defeated by the monster!");
                    break;
                }
            }
            System.out.println("\\nDo you want to play again? (y/n): ");
            String again = scanner.next();
            if (!again.equalsIgnoreCase("y")) {
                System.out.println("Thanks for playing! Goodbye!");
                break;
            }
        }
        scanner.close();
    }
}

def chatbot():
    print("Hello! I'm your chatbot. Type 'quit' to exit.")
    while True:
        user_input = input("> ").lower()
        if user_input == 'quit':
            print("Goodbye!")
            break
        elif "hello" in user_input:
            print("Hi there! How can I help you?")
        else:
            print("I don't understand, try again.")
chatbot()
    
# Linux shell script example
#!/bin/bash
echo "Hello! I'm your chatbot. Type 'quit' to exit."
while true; do
  read -p "> " user_input
  user_input=$(echo "$user_input" | tr '[:upper:]' '[:lower:]')
  if [ "$user_input" == "quit" ]; then
    echo "Goodbye!"
    break
  elif [[ "$user_input" == *"hello"* ]]; then
    echo "Hi there! How can I help you?"
  else
    echo "I don't understand, try again."
  fi
done
